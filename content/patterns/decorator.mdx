## Propósito

El **Decorator** permite agregar comportamientos a objetos individuales, sin afectar el comportamiento de otros objetos de la misma clase. Se logra envolviendo el objeto original en un objeto decorador que contiene el comportamiento adicional.

## Problema

La herencia es estática: no puedes alterar el comportamiento de una clase existente en tiempo de ejecución. Además, si necesitas combinar N características, necesitarías 2^N subclases.

## Solución

1. Asegúrate de que el componente base y los decoradores implementen la **misma interfaz**.
2. El decorador contiene una **referencia** al componente que envuelve (composición).
3. El decorador delega el trabajo al componente envuelto y puede añadir su propio comportamiento antes o después.

## Cuándo usar

- Cuando quieres añadir responsabilidades a objetos individuales de forma dinámica y transparente.
- Cuando la extensión por herencia no es práctica (demasiadas combinaciones posibles).
- Para aplicar responsabilidades que pueden ser retiradas posteriormente.

```typescript
// TypeScript — Middleware de request HTTP
interface RequestHandler {
  handle(request: { url: string; headers: Record<string, string>; body?: unknown }): Promise<Response>
}

class BaseHandler implements RequestHandler {
  async handle(request: { url: string; headers: Record<string, string> }): Promise<Response> {
    console.log(`Fetching ${request.url}`)
    return fetch(request.url, { headers: request.headers })
  }
}

abstract class HandlerDecorator implements RequestHandler {
  constructor(protected wrapped: RequestHandler) {}
  abstract handle(request: { url: string; headers: Record<string, string>; body?: unknown }): Promise<Response>
}

class AuthDecorator extends HandlerDecorator {
  constructor(wrapped: RequestHandler, private token: string) {
    super(wrapped)
  }

  async handle(request: { url: string; headers: Record<string, string> }): Promise<Response> {
    const withAuth = {
      ...request,
      headers: { ...request.headers, Authorization: `Bearer ${this.token}` },
    }
    return this.wrapped.handle(withAuth)
  }
}

class LoggingDecorator extends HandlerDecorator {
  async handle(request: { url: string; headers: Record<string, string> }): Promise<Response> {
    const start = Date.now()
    console.log(`[LOG] --> ${request.url}`)
    const response = await this.wrapped.handle(request)
    console.log(`[LOG] <-- ${request.url} (${Date.now() - start}ms)`)
    return response
  }
}

class RetryDecorator extends HandlerDecorator {
  constructor(wrapped: RequestHandler, private maxRetries: number) {
    super(wrapped)
  }

  async handle(request: { url: string; headers: Record<string, string> }): Promise<Response> {
    for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
      try {
        return await this.wrapped.handle(request)
      } catch (err) {
        if (attempt === this.maxRetries) throw err
        console.log(`Retry ${attempt + 1}/${this.maxRetries}...`)
      }
    }
    throw new Error('Max retries exceeded')
  }
}

// Uso — composición de decoradores
const handler = new LoggingDecorator(
  new RetryDecorator(
    new AuthDecorator(new BaseHandler(), 'my-secret-token'),
    3,
  ),
)

handler.handle({ url: 'https://api.example.com/data', headers: {} })
```

```python
# Python — Decoradores de validación y transformación de datos
from functools import wraps
from typing import Any, Callable
import time

def timer(func: Callable) -> Callable:
    @wraps(func)
    def wrapper(*args: Any, **kwargs: Any) -> Any:
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f'{func.__name__} took {elapsed:.4f}s')
        return result
    return wrapper

def validate_positive(func: Callable) -> Callable:
    @wraps(func)
    def wrapper(value: float, *args: Any, **kwargs: Any) -> Any:
        if value <= 0:
            raise ValueError(f'Expected positive value, got {value}')
        return func(value, *args, **kwargs)
    return wrapper

def cache(func: Callable) -> Callable:
    memo: dict = {}
    @wraps(func)
    def wrapper(*args: Any) -> Any:
        if args not in memo:
            memo[args] = func(*args)
        return memo[args]
    return wrapper

@timer
@validate_positive
@cache
def compute_factorial(n: int) -> int:
    if n <= 1:
        return 1
    return n * compute_factorial(n - 1)

print(compute_factorial(10))
```

```go
// Go — Decoradores de middleware HTTP
package decorator

import (
    "fmt"
    "net/http"
    "time"
)

type Middleware func(http.Handler) http.Handler

func Logger(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        next.ServeHTTP(w, r)
        fmt.Printf("[%s] %s %s\n", time.Since(start), r.Method, r.URL.Path)
    })
}

func Auth(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("Authorization")
        if token == "" {
            http.Error(w, "Unauthorized", http.StatusUnauthorized)
            return
        }
        next.ServeHTTP(w, r)
    })
}

func Chain(h http.Handler, middlewares ...Middleware) http.Handler {
    for i := len(middlewares) - 1; i >= 0; i-- {
        h = middlewares[i](h)
    }
    return h
}
```

```java
// Java — Streams decorados
import java.util.function.Function;
import java.util.function.UnaryOperator;

public class TextProcessor {
    private UnaryOperator<String> pipeline = UnaryOperator.identity();

    public TextProcessor addStep(UnaryOperator<String> step) {
        pipeline = pipeline.andThen(step)::apply;
        return this;
    }

    public String process(String input) {
        return pipeline.apply(input);
    }
}

// Uso
var processor = new TextProcessor()
    .addStep(String::trim)
    .addStep(String::toLowerCase)
    .addStep(s -> s.replaceAll("\\s+", "-"))
    .addStep(s -> "slug: " + s);

System.out.println(processor.process("  Hello World  ")); // slug: hello-world
```

## Patrones relacionados

- **Composite** — comparte la estructura de árbol, pero su propósito es diferente
- **Strategy** — cambia el interior del objeto; Decorator cambia su piel
- **Chain of Responsibility** — el handler puede pasarse a sí mismo como parte del procesamiento
