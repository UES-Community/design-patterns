## Propósito

El **Facade** proporciona una interfaz de alto nivel simplificada a un conjunto complejo de subsistemas. Hace que el subsistema sea más fácil de usar sin exponerlo completamente.

## Cuándo usar

- Cuando quieres proporcionar una interfaz simple a un subsistema complejo.
- Cuando hay muchas dependencias entre clientes y las clases de implementación.
- Cuando quieres organizar un subsistema en capas.

```typescript
// TypeScript — Facade para procesamiento de video
class VideoDecoder {
  decode(buffer: Buffer): ArrayBuffer {
    console.log('Decoding video...')
    return buffer.buffer
  }
}

class AudioDecoder {
  decode(buffer: Buffer): Float32Array {
    console.log('Decoding audio...')
    return new Float32Array(buffer.buffer)
  }
}

class VideoEncoder {
  encode(frames: ArrayBuffer, format: string): Buffer {
    console.log(`Encoding to ${format}...`)
    return Buffer.from(frames)
  }
}

class AudioMixer {
  mix(tracks: Float32Array[]): Float32Array {
    console.log(`Mixing ${tracks.length} audio tracks...`)
    return tracks[0]
  }
}

class MetadataWriter {
  write(output: Buffer, meta: Record<string, unknown>): Buffer {
    console.log('Writing metadata...')
    return output
  }
}

// Facade
class VideoConverter {
  private decoder = new VideoDecoder()
  private audioDecoder = new AudioDecoder()
  private encoder = new VideoEncoder()
  private mixer = new AudioMixer()
  private metaWriter = new MetadataWriter()

  convert(input: Buffer, format: 'mp4' | 'webm', meta?: Record<string, unknown>): Buffer {
    const videoFrames = this.decoder.decode(input)
    const audioTrack = this.audioDecoder.decode(input)
    const mixedAudio = this.mixer.mix([audioTrack])
    const encoded = this.encoder.encode(videoFrames, format)
    return meta ? this.metaWriter.write(encoded, meta) : encoded
  }
}

// Uso — una sola llamada en lugar de coordinar 5 subsistemas
const converter = new VideoConverter()
converter.convert(Buffer.alloc(0), 'mp4', { title: 'My Video' })
```

```python
# Python — Facade para sistema de pedidos e-commerce
class InventoryService:
    def check_stock(self, product_id: str, quantity: int) -> bool:
        print(f'Checking stock for {product_id}')
        return True

    def reserve(self, product_id: str, quantity: int) -> str:
        return f'RESERVATION-{product_id}'

class PaymentService:
    def charge(self, card_token: str, amount: float) -> str:
        print(f'Charging ${amount}')
        return f'PAYMENT-{id(self)}'

class ShippingService:
    def calculate_cost(self, destination: str) -> float:
        return 5.99

    def create_label(self, order_id: str, destination: str) -> str:
        return f'LABEL-{order_id}'

class NotificationService:
    def send_confirmation(self, email: str, order_id: str) -> None:
        print(f'Sending confirmation to {email}')

class OrderFacade:
    def __init__(self):
        self._inventory = InventoryService()
        self._payment = PaymentService()
        self._shipping = ShippingService()
        self._notifications = NotificationService()

    def place_order(self, product_id: str, quantity: int,
                    card_token: str, email: str, destination: str) -> dict:
        if not self._inventory.check_stock(product_id, quantity):
            raise ValueError('Out of stock')

        reservation = self._inventory.reserve(product_id, quantity)
        shipping_cost = self._shipping.calculate_cost(destination)
        payment_id = self._payment.charge(card_token, 49.99 + shipping_cost)
        label = self._shipping.create_label(payment_id, destination)
        self._notifications.send_confirmation(email, payment_id)

        return {'order_id': payment_id, 'label': label, 'reservation': reservation}

# Uso simple
facade = OrderFacade()
order = facade.place_order('PROD-1', 2, 'tok_visa', 'user@example.com', 'NYC')
```

```go
// Go — Facade para autenticación
package facade

type TokenService struct{}
func (t *TokenService) Generate(userID string) string { return "jwt_token" }
func (t *TokenService) Validate(token string) (string, error) { return "user_id", nil }

type SessionStore struct{}
func (s *SessionStore) Create(userID string) string { return "session_id" }
func (s *SessionStore) Destroy(sessionID string) {}

type AuditLogger struct{}
func (a *AuditLogger) Log(event, userID string) {}

type AuthFacade struct {
    tokens   *TokenService
    sessions *SessionStore
    audit    *AuditLogger
}

func NewAuthFacade() *AuthFacade {
    return &AuthFacade{&TokenService{}, &SessionStore{}, &AuditLogger{}}
}

func (a *AuthFacade) Login(userID string) (token, sessionID string) {
    token = a.tokens.Generate(userID)
    sessionID = a.sessions.Create(userID)
    a.audit.Log("login", userID)
    return
}
```

```java
// Java — Facade para sistema de reportes
public class ReportFacade {
    private final DataFetcher fetcher = new DataFetcher();
    private final DataTransformer transformer = new DataTransformer();
    private final ReportRenderer renderer = new ReportRenderer();
    private final PDFExporter exporter = new PDFExporter();

    public byte[] generatePDFReport(String reportType, String dateRange) {
        var raw = fetcher.fetch(reportType, dateRange);
        var data = transformer.transform(raw);
        var report = renderer.render(data, reportType);
        return exporter.export(report);
    }
}
```
