## Propósito

El **Command** convierte una solicitud en un objeto independiente que contiene toda la información sobre esa solicitud. Esta transformación permite pasar solicitudes como argumentos, retrasarlas, ponerlas en cola, y soportar operaciones deshacer/rehacer.

## Problema

Tienes botones, atajos de teclado y opciones de menú que necesitan ejecutar las mismas operaciones. Además, necesitas historial de acciones para implementar Ctrl+Z.

## Cuándo usar

- Para implementar operaciones reversibles (undo/redo).
- Para parametrizar objetos con operaciones.
- Para implementar cambios diferidos o colas de operaciones.
- Para implementar logging de operaciones para auditoría.

```typescript
// TypeScript — Editor de texto con undo/redo
interface Command {
  execute(): void
  undo(): void
  description: string
}

class TextEditor {
  private content = ''
  private cursor = 0

  insert(text: string, position: number): void {
    this.content = this.content.slice(0, position) + text + this.content.slice(position)
    this.cursor = position + text.length
  }

  delete(start: number, length: number): void {
    this.content = this.content.slice(0, start) + this.content.slice(start + length)
    this.cursor = start
  }

  getContent(): string { return this.content }
  getCursor(): number { return this.cursor }
}

class InsertCommand implements Command {
  description: string

  constructor(
    private editor: TextEditor,
    private text: string,
    private position: number,
  ) {
    this.description = `Insert "${text}" at ${position}`
  }

  execute(): void {
    this.editor.insert(this.text, this.position)
  }

  undo(): void {
    this.editor.delete(this.position, this.text.length)
  }
}

class DeleteCommand implements Command {
  private deletedText = ''
  description: string

  constructor(
    private editor: TextEditor,
    private start: number,
    private length: number,
  ) {
    this.description = `Delete ${length} chars at ${start}`
  }

  execute(): void {
    this.deletedText = this.editor.getContent().slice(this.start, this.start + this.length)
    this.editor.delete(this.start, this.length)
  }

  undo(): void {
    this.editor.insert(this.deletedText, this.start)
  }
}

class CommandHistory {
  private history: Command[] = []
  private redoStack: Command[] = []

  execute(command: Command): void {
    command.execute()
    this.history.push(command)
    this.redoStack = []
    console.log(`Executed: ${command.description}`)
  }

  undo(): void {
    const command = this.history.pop()
    if (!command) return
    command.undo()
    this.redoStack.push(command)
    console.log(`Undone: ${command.description}`)
  }

  redo(): void {
    const command = this.redoStack.pop()
    if (!command) return
    command.execute()
    this.history.push(command)
    console.log(`Redone: ${command.description}`)
  }
}

// Uso
const editor = new TextEditor()
const history = new CommandHistory()

history.execute(new InsertCommand(editor, 'Hello', 0))
history.execute(new InsertCommand(editor, ' World', 5))
console.log(editor.getContent()) // "Hello World"

history.undo()
console.log(editor.getContent()) // "Hello"

history.redo()
console.log(editor.getContent()) // "Hello World"
```

```python
# Python — Sistema de transacciones bancarias
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Optional

@dataclass
class Account:
    id: str
    balance: float = 0.0

    def deposit(self, amount: float) -> None:
        self.balance += amount

    def withdraw(self, amount: float) -> None:
        if amount > self.balance:
            raise ValueError(f'Insufficient funds: {self.balance}')
        self.balance -= amount

class BankCommand(ABC):
    @abstractmethod
    def execute(self) -> None: ...

    @abstractmethod
    def undo(self) -> None: ...

class DepositCommand(BankCommand):
    def __init__(self, account: Account, amount: float):
        self.account = account
        self.amount = amount

    def execute(self) -> None:
        self.account.deposit(self.amount)
        print(f'Deposited ${self.amount} → Balance: ${self.account.balance}')

    def undo(self) -> None:
        self.account.withdraw(self.amount)
        print(f'Reversed deposit of ${self.amount}')

class TransactionManager:
    def __init__(self):
        self._history: list[BankCommand] = []

    def execute(self, command: BankCommand) -> None:
        command.execute()
        self._history.append(command)

    def rollback_last(self) -> None:
        if self._history:
            self._history.pop().undo()

# Uso
account = Account('ACC-001', 1000.0)
manager = TransactionManager()
manager.execute(DepositCommand(account, 500.0))
manager.rollback_last()
print(f'Final balance: ${account.balance}')
```

```go
// Go — Task queue con comandos serializables
package command

type Command interface {
    Execute() error
    Undo() error
    Name() string
}

type Queue struct {
    commands []Command
    executed []Command
}

func (q *Queue) Enqueue(cmd Command) {
    q.commands = append(q.commands, cmd)
}

func (q *Queue) Run() error {
    for len(q.commands) > 0 {
        cmd := q.commands[0]
        q.commands = q.commands[1:]
        if err := cmd.Execute(); err != nil {
            return err
        }
        q.executed = append(q.executed, cmd)
    }
    return nil
}

func (q *Queue) UndoLast() error {
    if len(q.executed) == 0 {
        return nil
    }
    last := q.executed[len(q.executed)-1]
    q.executed = q.executed[:len(q.executed)-1]
    return last.Undo()
}
```

```java
// Java — GUI con macro recording
import java.util.ArrayDeque;
import java.util.Deque;

public interface Command {
    void execute();
    void undo();
}

public class MacroRecorder {
    private final Deque<Command> history = new ArrayDeque<>();
    private boolean recording = false;
    private final java.util.List<Command> macro = new java.util.ArrayList<>();

    public void execute(Command cmd) {
        cmd.execute();
        history.push(cmd);
        if (recording) macro.add(cmd);
    }

    public void undo() {
        if (!history.isEmpty()) history.pop().undo();
    }

    public void startRecording() { recording = true; macro.clear(); }
    public void stopRecording()  { recording = false; }
    public void playMacro()      { macro.forEach(Command::execute); }
}
```

## Patrones relacionados

- **Chain of Responsibility** — pasa la solicitud por una cadena; Command encapsula la solicitud
- **Memento** — puede usarse junto con Command para almacenar el estado antes de ejecutar
- **Strategy** — similar, pero Strategy reemplaza parte del objeto; Command representa la solicitud completa
