```python
# Python — Adaptar loggers de distintas librerías
import logging
from abc import ABC, abstractmethod

class Logger(ABC):
    @abstractmethod
    def log(self, level: str, message: str) -> None: ...

class PythonLoggerAdapter(Logger):
    def __init__(self, name: str):
        self._logger = logging.getLogger(name)
        logging.basicConfig(level=logging.DEBUG)

    def log(self, level: str, message: str) -> None:
        getattr(self._logger, level.lower(), self._logger.info)(message)

class PrintLogger:
    """Logger legacy que solo imprime"""
    def write(self, text: str) -> None:
        print(f'LEGACY: {text}')

class PrintLoggerAdapter(Logger):
    def __init__(self, legacy: PrintLogger):
        self._legacy = legacy

    def log(self, level: str, message: str) -> None:
        self._legacy.write(f'[{level}] {message}')

# Uso unificado
def process_data(logger: Logger, data: list) -> None:
    logger.log('info', f'Processing {len(data)} items')
    for item in data:
        logger.log('debug', f'Item: {item}')

process_data(PythonLoggerAdapter('app'), [1, 2, 3])
process_data(PrintLoggerAdapter(PrintLogger()), [4, 5, 6])
```
