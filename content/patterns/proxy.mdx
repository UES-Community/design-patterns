## Propósito

El **Proxy** actúa como un intermediario entre el cliente y el objeto real. Puede añadir funcionalidad (caché, control de acceso, logging) sin modificar la clase original.

## Tipos de Proxy

- **Virtual Proxy** — retrasa la creación/carga de objetos costosos hasta que son necesarios (lazy loading).
- **Protection Proxy** — controla el acceso al objeto original según los permisos.
- **Caching Proxy** — guarda los resultados de operaciones costosas para reutilizarlos.
- **Remote Proxy** — representa un objeto en otro espacio de direcciones (RPC).

## Cuándo usar

- Para controlar el acceso a un objeto (autenticación, autorización).
- Para implementar lazy initialization de objetos costosos.
- Para implementar caché de resultados.
- Para logging o auditoría transparente.

```typescript
// TypeScript — Proxy de caché con TTL
interface DataService {
  fetchUser(id: string): Promise<{ id: string; name: string; email: string }>
  fetchPosts(userId: string): Promise<Array<{ id: string; title: string }>>
}

class RealDataService implements DataService {
  async fetchUser(id: string) {
    console.log(`[DB] Fetching user ${id}...`)
    await new Promise(r => setTimeout(r, 100)) // simula latencia
    return { id, name: `User ${id}`, email: `user${id}@example.com` }
  }

  async fetchPosts(userId: string) {
    console.log(`[DB] Fetching posts for ${userId}...`)
    await new Promise(r => setTimeout(r, 150))
    return [{ id: '1', title: 'Post 1' }, { id: '2', title: 'Post 2' }]
  }
}

interface CacheEntry<T> {
  value: T
  expiresAt: number
}

class CachingProxy implements DataService {
  private cache = new Map<string, CacheEntry<unknown>>()

  constructor(
    private real: DataService,
    private ttlMs = 60_000,
  ) {}

  private getFromCache<T>(key: string): T | null {
    const entry = this.cache.get(key)
    if (!entry) return null
    if (Date.now() > entry.expiresAt) {
      this.cache.delete(key)
      return null
    }
    console.log(`[CACHE HIT] ${key}`)
    return entry.value as T
  }

  private setCache<T>(key: string, value: T): void {
    this.cache.set(key, { value, expiresAt: Date.now() + this.ttlMs })
  }

  async fetchUser(id: string) {
    const key = `user:${id}`
    const cached = this.getFromCache<Awaited<ReturnType<DataService['fetchUser']>>>(key)
    if (cached) return cached
    const result = await this.real.fetchUser(id)
    this.setCache(key, result)
    return result
  }

  async fetchPosts(userId: string) {
    const key = `posts:${userId}`
    const cached = this.getFromCache<Awaited<ReturnType<DataService['fetchPosts']>>>(key)
    if (cached) return cached
    const result = await this.real.fetchPosts(userId)
    this.setCache(key, result)
    return result
  }
}

// Uso
const service = new CachingProxy(new RealDataService())
await service.fetchUser('42')   // [DB] consulta real
await service.fetchUser('42')   // [CACHE HIT]
```

```python
# Python — Protection Proxy con RBAC
from abc import ABC, abstractmethod
from enum import Enum
from functools import wraps

class Role(Enum):
    GUEST = 'guest'
    USER = 'user'
    ADMIN = 'admin'

class FileSystem(ABC):
    @abstractmethod
    def read(self, path: str) -> str: ...

    @abstractmethod
    def write(self, path: str, content: str) -> None: ...

    @abstractmethod
    def delete(self, path: str) -> None: ...

class RealFileSystem(FileSystem):
    def read(self, path: str) -> str:
        return f'Content of {path}'

    def write(self, path: str, content: str) -> None:
        print(f'Writing to {path}')

    def delete(self, path: str) -> None:
        print(f'Deleting {path}')

class ProtectedFileSystem(FileSystem):
    PERMISSIONS = {
        Role.GUEST: {'read'},
        Role.USER:  {'read', 'write'},
        Role.ADMIN: {'read', 'write', 'delete'},
    }

    def __init__(self, real: RealFileSystem, role: Role):
        self._real = real
        self._role = role

    def _check(self, operation: str) -> None:
        if operation not in self.PERMISSIONS.get(self._role, set()):
            raise PermissionError(f'{self._role.value} cannot {operation}')

    def read(self, path: str) -> str:
        self._check('read')
        return self._real.read(path)

    def write(self, path: str, content: str) -> None:
        self._check('write')
        self._real.write(path, content)

    def delete(self, path: str) -> None:
        self._check('delete')
        self._real.delete(path)

# Uso
fs = ProtectedFileSystem(RealFileSystem(), Role.USER)
print(fs.read('/etc/config'))   # OK
fs.write('/tmp/data', 'hello')  # OK
fs.delete('/important')         # PermissionError
```

```go
// Go — Lazy loading proxy
package proxy

import "sync"

type Image interface {
    Display() string
    Size() int64
}

type HighResImage struct {
    filename string
    data     []byte
}

func (h *HighResImage) Display() string { return "Displaying " + h.filename }
func (h *HighResImage) Size() int64     { return int64(len(h.data)) }

type LazyImage struct {
    filename string
    real     *HighResImage
    once     sync.Once
}

func (l *LazyImage) load() {
    l.once.Do(func() {
        println("Loading image: " + l.filename)
        l.real = &HighResImage{filename: l.filename, data: make([]byte, 1024)}
    })
}

func (l *LazyImage) Display() string { l.load(); return l.real.Display() }
func (l *LazyImage) Size() int64     { l.load(); return l.real.Size() }
```

```java
// Java — Proxy dinámico para logging
import java.lang.reflect.*;

public class LoggingProxy {
    @SuppressWarnings("unchecked")
    public static <T> T create(T target, Class<T> iface) {
        return (T) Proxy.newProxyInstance(
            iface.getClassLoader(),
            new Class[]{iface},
            (proxy, method, args) -> {
                System.out.println("--> " + method.getName());
                Object result = method.invoke(target, args);
                System.out.println("<-- " + method.getName());
                return result;
            }
        );
    }
}
```
